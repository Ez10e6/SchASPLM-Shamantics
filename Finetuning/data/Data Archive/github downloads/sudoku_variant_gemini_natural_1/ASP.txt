% Define the domain of values from 1 to 9.
value(1..9).

% Generate a solution where every cell (Row, Column) is assigned exactly one Value.
1 { sudoku(R, C, V) : value(V) } 1 :- value(R), value(C).

% Constraint: Each row must contain exactly one assignment for every Value.
1 { sudoku(R, C, V) : value(C) } 1 :- value(R), value(V).

% Constraint: Each column must contain exactly one assignment for every Value.
1 { sudoku(R, C, V) : value(R) } 1 :- value(C), value(V).

% Constraint: No two distinct cells within the same 3x3 subgrid can contain the same Value.
:- sudoku(R, C, V), sudoku(S, D, V), (R,C) != (S,D), (R-1)/3 == (S-1)/3, (C-1)/3 == (D-1)/3.
