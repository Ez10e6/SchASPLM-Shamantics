% Marriage is a symmetric relation (if A is married to B, B is married to A).
married(Bride, Groom) :- married(Groom, Bride), Bride != Groom.

% Define siblings as distinct individuals who share a parent.
sibling(Child1, Child2) :- parent(Parent, Child1), parent(Parent, Child2), Child1 != Child2.

% Define stepparents: a spouse of a parent is also considered a parent.
parent(Stepparent, Child) :- parent(Parent, Child), married(Parent, Stepparent).

% Define uncles as male siblings of a parent.
uncle(Uncle, Child) :- sibling(Parent, Uncle), parent(Parent, Child), male(Uncle).

% Define grandparents as parents of a parent.
grandparent(Grandparent, Child) :- parent(Grandparent, Parent), parent(Parent, Child).

% Define a 'grandpa' as a male who is a grandparent of themselves.
grandpa(Person) :- grandparent(Person, Person), male(Person).
