% Select exactly one data plan from {BasicData, PremiumData}.
plan("BasicData"; "PremiumData").
1 { selected_plan(P) : plan(P) } 1.

% Select exactly one phone model from {ModelX, ModelY}.
phone("ModelX"; "ModelY").
1 { selected_phone(M) : phone(M) } 1.

% Select exactly one accessory from {CaseA, CaseB}.
accessory("CaseA"; "CaseB").
1 { selected_accessory(A) : accessory(A) } 1.

% BasicData is not compatible with ModelY.
:- selected_plan("BasicData"), selected_phone("ModelY").

% ModelX requires CaseA.
:- selected_phone("ModelX"), selected_accessory(A), A != "CaseA".

% PremiumData cannot be used with CaseA.
:- selected_plan("PremiumData"), selected_accessory("CaseA").
