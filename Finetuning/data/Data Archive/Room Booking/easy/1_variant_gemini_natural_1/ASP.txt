% There are 6 seminars
seminar(1..6).

% There are 3 time slots: morning, midday, afternoon.
timeslot("morning"; "midday"; "afternoon").

% There are 3 rooms
room(1; 2; 3).

% All seminars must be delivered
1{delivered(S, R, T) : room(R), timeslot(T)} :- seminar(S).

% Only one seminar can be scheduled in a room at a time.
:- delivered(S1, R, T), delivered(S2, R, T), S1 != S2.

% Seminars can not be delivered in the afternoon in room 3.
:- schedule(S, 3, afternoon).

% Seminar 1 must be in Room 1.
:- schedule(1, R, T), R != 1.

% Seminars 2 and 3 must not be at the same time.
:- delivered(2, _, T), delivered(3, _, T).
