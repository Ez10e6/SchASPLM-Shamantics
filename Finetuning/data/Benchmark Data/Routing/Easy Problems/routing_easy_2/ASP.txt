% 4 Locations
loc(1..4).

% Adjacent nodes: (1–2), (1–3), (2–4)
edge(1, 2).
edge(1, 3).
edge(2, 4).

% The drone can only move between adjacent nodes
1 { at(L, T) : loc(L) } 1 :- time(T).
:- at(L1, T), at(L2, T+1), not edge(L1, L2), not edge(L2, L1), time(T).

% There are 5 timesteps
time(0..5).

% Must visit locations: 3, 4
at(3, _).
at(4, _).

% The drone starts at 1
at(1, 0).
