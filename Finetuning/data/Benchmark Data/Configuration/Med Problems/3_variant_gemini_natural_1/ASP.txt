% Frames: Lightweight (Endurance 50), Standard (Endurance 40), Reinforced (Endurance 30)
frame("Lightweight", 50; "Standard", 40; "Reinforced", 30).

% Motors: Eco (Endurance 60), Standard (Endurance 70), HighTorque (Endurance 90)
motor("Eco", 60; "Standard", 70; "HighTorque", 90).

% Batteries: SmallCell (Endurance 30), MediumCell (Endurance 50), LongLife (Endurance 80)
battery("SmallCell", 30; "MediumCell", 50; "LongLife", 80).

% Select one frame, one motor, one battery
1 { selected_frame(F) : frame(F, _) } 1.
1 { selected_motor(M) : motor(M, _) } 1.
1 { selected_battery(B) : battery(B, _) } 1.

% HighTorque requires LongLife battery
:- selected_motor("HighTorque"), selected_battery(B), B != "LongLife".

% Lightweight frame is incompatible with HighTorque motor.
:- selected_frame("Lightweight"), selected_motor("HighTorque").

% LongLife battery requires Reinforced frame
:- selected_battery("LongLife"), selected_frame(F), F != "Reinforced".

% SmallCell battery only works with Eco motor.
:- selected_battery("SmallCell"), selected_motor(M), M != "Eco".

% MediumCell battery supports only Eco or Standard motor
:- selected_battery("MediumCell"), selected_motor("HighTorque").

% Total endurance score
total_endurance(E) :-
    selected_frame(F), frame(F, EF),
    selected_motor(M), motor(M, EM),
    selected_battery(B), battery(B, EB),
    E = EF + EM + EB.
#maximize { E : total_endurance(E) }.
